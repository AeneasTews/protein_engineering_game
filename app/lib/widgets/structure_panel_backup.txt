import 'package:flutter/material.dart';
import 'package:flutter_bloc/flutter_bloc.dart';
import 'package:webview_flutter/webview_flutter.dart';
import '../../blocs/structure/structure_bloc.dart';
import '../presentation/molstar_controller.dart';
import '../app_theme.dart';

class StructurePanel extends StatefulWidget {
  const StructurePanel({super.key});

  @override
  State<StructurePanel> createState() => _StructurePanelState();
}

class _StructurePanelState extends State<StructurePanel> {
  late final MolstarController _molstar;

  @override
  void initState() {
    super.initState();
    _molstar = MolstarController();

    // Forward residue-click events from Mol* into StructureBloc.
    // MolstarController doesn't know about blocs — it just calls this function.
    _molstar.onResidueSelected = (position) {
      if (mounted) {
        context.read<StructureBloc>().add(ResidueSelected(position: position));
        // Also sync the selection back into ExperimentBloc so the AA picker
        // can open on the correct residue in the sequence panel.
      }
    };
  }

  @override
  Widget build(BuildContext context) {
    return Stack(
      children: [
        // The WebView fills the entire panel
        WebViewWidget(controller: _molstar.webviewController),

        // BlocListener handles all state-driven Mol* commands.
        // Using a listener (not a builder) because we're driving side effects,
        // not rebuilding widgets.
        BlocListener<StructureBloc, StructureState>(
          listener: (context, state) async {
            if (state is StructureReady) {
              await _molstar.highlightResidues(state.highlightedMutations);

              if (state.selectedPosition != null) {
                await _molstar.focusResidue(state.selectedPosition!);
              } else {
                await _molstar.clearHighlights();
              }
            }
          },
          child: const SizedBox.shrink(),
        ),

        // Separate listener for the load event — only fires once
        BlocListener<StructureBloc, StructureState>(
          listenWhen: (prev, next) =>
          prev is StructureInitial && next is StructureReady,
          listener: (context, state) async {
            // The PDB data was already downloaded by GameScreen._loadStructure()
            // and stored... but we need it here. We retrieve it from the
            // ExperimentBloc's protein — actually we need the raw PDB string.
            // GameScreen passes it via StructureLoadRequested which carries pdbData.
            // We stored it nowhere accessible here, so we read it from the event.
            // The cleanest fix: store pdbData in StructureReady state.
            // For now, StructureLoadRequested carries the data and the bloc
            // passes it through — see note below.
          },
          child: const SizedBox.shrink(),
        ),

        // Overlay: shown while the structure is still loading
        BlocBuilder<StructureBloc, StructureState>(
          builder: (context, state) {
            if (state is StructureInitial) {
              return Container(
                color: AppTheme.background,
                child: const Center(
                  child: Column(
                    mainAxisSize: MainAxisSize.min,
                    children: [
                      CircularProgressIndicator(color: AppTheme.accent),
                      SizedBox(height: 16),
                      Text('Loading structure...',
                          style: TextStyle(color: AppTheme.textSecondary)),
                    ],
                  ),
                ),
              );
            }
            return const SizedBox.shrink();
          },
        ),
      ],
    );
  }
}
